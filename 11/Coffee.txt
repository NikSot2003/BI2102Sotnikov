// Абстрактный класс для всех напитков, содержащий общие параметры: вода, зёрна и молоко.
abstract class Coffee { // Определение абстрактного класса Coffee, который будет использоваться как базовый для всех видов напитков.
  int water; // Количество воды, необходимое для напитка (мл).
  int beans; // Количество зёрен, необходимое для напитка (г).
  int milk; // Количество молока, необходимое для напитка (мл).
  double price; // Цена напитка.

  // Конструктор, инициализирующий параметры напитка.
  Coffee(this.water, this.beans, this.milk, this.price); // Конструктор принимает параметры и инициализирует их для каждого экземпляра напитка.
}

// Класс Эспрессо, наследующий Coffee, с конкретными значениями ресурсов.
class Espresso extends Coffee { // Класс для напитка Эспрессо, который наследует абстрактный класс Coffee.
  Espresso() : super(30, 10, 0, 2.5); // В конструкторе вызывается конструктор родительского класса (super), инициализируя значения: 30 мл воды, 10 г зёрен, 0 мл молока, 2.5 единиц цены.
}

// Класс Американо, наследующий Coffee, с конкретными значениями ресурсов.
class Americano extends Coffee { // Класс для напитка Американо, который также наследует от Coffee.
  Americano() : super(150, 10, 0, 3.0); // В конструкторе передаются параметры: 150 мл воды, 10 г зёрен, 0 мл молока, 3.0 единиц цены.
}

// Класс Латте, наследующий Coffee, с конкретными значениями ресурсов и молоком.
class Latte extends Coffee { // Класс для напитка Латте, который наследует от Coffee.
  Latte() : super(150, 10, 100, 4.0); // В конструкторе передаются параметры: 150 мл воды, 10 г зёрен, 100 мл молока, 4.0 единиц цены.
}
